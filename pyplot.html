# Generate a list of at least 5 random integers
random_numbers = [randint(10, 100) for _ in range(7)]
indices = list(range(len(random_numbers)))

# Set up subplots
fig, (ax1, ax2) = plt.subplots(2, 1, figsize=(8, 6))

#  SCATTER PLOT with customized style
ax1.scatter(
    indices,
    random_numbers,
    color='crimson',       # Color of the points
    marker='^',            # Triangle marker
    s=100,                 # Marker size
    edgecolor='black'      # Border around markers
)
ax1.set_title("Styled Scatter Plot of Random Numbers", fontsize=14, color='darkred')
ax1.set_xlabel("Index", fontsize=12)
ax1.set_ylabel("Value", fontsize=12)
ax1.grid(True, linestyle='--', alpha=0.5)

#  LINE PLOT with customized style
ax2.plot(
    indices,
    random_numbers,
    color='mediumseagreen',
    marker='D',            # Diamond marker
    linestyle='--',        # Dashed line
    linewidth=2,
    markersize=8,
    markerfacecolor='yellow',
